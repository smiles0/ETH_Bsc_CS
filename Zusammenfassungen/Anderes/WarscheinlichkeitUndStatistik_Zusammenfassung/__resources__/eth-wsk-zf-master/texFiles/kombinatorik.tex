\section{Kombinatorik}

\subsection{Permutation}
n unterschiedlichen Kugeln:
\[
P(n)=n!
\]

n unterschiedliche Kugeln mit $n_1,n_2,...,n_k$ gleichen Kugeln:
\[
P(n,n_1,n_2,...,n_k)=\frac{n!}{n_1!n_2!...n_k!}
\]

\textbf{Bemerkungen:}
\begin{itemize}
\item Ermitteln der Anzahl Anordnungen.
\end{itemize}
